
import unittest
from employee import Employee

class TestEmployee(unittest.TestCase):

    @classmethod
    def setUpClass(cls):
        print('setup class ')
    @classmethod
    def tearDownClass(cls):
        print('teardown class ')
    #called before every testcase

    def setUp(self):
        print('called before')
        self.emp1= Employee('Corey','schafer',50000)
        self.emp2= Employee('Sue','Smith',60000)
    def tearDown(self):
        print('called after')
        pass

    def test_email(self):
        print('testing email')
        self.assertEqual(self.emp1.email,'Corey.schafer@email.com')
        self.assertEqual(self.emp2.email,'Sue.Smith@email.com')
    
    def test_fullname(self):
         print('testing fullname')
         emp1= Employee('Corey','schafer',50000)
         self.assertEqual(emp1.fullname,'Corey schafer')

    def test_apply_raise(self):
         print('testing salary')
         emp1= Employee('Corey','schafer',50000)
         emp1.apply_raise()
         self.assertEqual(emp1.pay,52500)

if __name__ == '__main__':
    unittest.main()

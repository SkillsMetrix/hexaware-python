
class Employee:
 

    def __init__(self,name,email,salary):
        self.name=name
        self.salary=salary
        self.email=email
    
    def __repr__(self):
        return f"Employee( '{self.name}' , '{self.email}' , '{self.salary})'"
    
    @property
    def email(self):
        return f"{self.name.lower()}@hexaware.com"
    @property
    def name(self):
        return f"{self.name}"
    @property
    def salary(self):
        return f"{self.salary}"
    @salary.setter
    def salary(self,value):
        if value < 0:
            raise ValueError("Salary must  be non-negative")
        self.salary=value

emp=Employee('user1','user@mail.com',3000)

print(repr(emp))
print(emp.email)
emp.salary=4500
print(emp.salary)
